INCLUDE_DIRECTORIES(${LBCpp_SOURCE_DIR}/include)

MACRO(OPTIONAL_PROJECT NAME DIRECTORY)
  OPTION(LBCPP_PROJECT_${NAME} "Include project ${DIRECTORY}")
  IF(LBCPP_PROJECT_${NAME})
    ADD_SUBDIRECTORY(${DIRECTORY})

  ENDIF(LBCPP_PROJECT_${NAME})
ENDMACRO(OPTIONAL_PROJECT)

OPTIONAL_PROJECT(PROTEINS Proteins)
OPTIONAL_PROJECT(NUMBERS Numbers)
OPTIONAL_PROJECT(MNIST MNIST)
OPTIONAL_PROJECT(NETWORK Network)
